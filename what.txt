  <div id="grid"></div>
  <script>
    fetch('games.json')
      .then(r=>r.json())
      .then(draw);
    function draw(list){
      const grid=document.getElementById('grid'),search=document.getElementById('search');
      function render(filter=''){
        grid.innerHTML='';
        list.filter(g=>g.title.toLowerCase().includes(filter.toLowerCase()))
            .forEach(g=>{
              const card=document.createElement('a');
              card.className='card'; card.href=g.file; card.target='_blank';
              card.innerHTML=`<img src="${g.thumb}" onerror="this.src='assets/placeholder.png'"/><span>${g.title}</span>`;
              grid.appendChild(card);
            });
      }
      render();
      search.addEventListener('input',e=>render(e.target.value));
    }
  </script>
